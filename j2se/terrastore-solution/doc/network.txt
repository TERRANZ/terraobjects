Протокол взаимодействия
int - опкод, в зависимости от опкода читаем дальше

login: 
int длина пароля
byte[] пароль в UTF-8
возврат: int ServerOpcodes.LOGIN_FAILED или ServerOpcodes.LOGIN_OK
int OK

GET_AVAIL_OBJECTS:
return: int[] of avail templates
int OK

READY:
меняет состояние сервера, обозначает, что клиент слушает сервер
ret: int OK

GET_OBJECTS:
int - templateId
return: 
int - count of objects for send
(
int - count of props
int - prop type
в зависимости от типа
str: int len, byte[] str
int: int values
float: float value
text: same as str
date: long date value
) 
int OK

NEW_OBJ:
int templateId
int parentId
ret: int OK

SET_PROP_VAL:
int objId
int propId
int propType
в зависимости от типа:
str: int len, byte[] val
int: int value
float: float value
text: same as str
date: long value
ret: int OK

SET_PROPS_VAL:
int count
for (count) SET_PROP_VAL
